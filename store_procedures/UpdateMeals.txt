CREATE OR REPLACE PROCEDURE SP_UPDATE_MEALS(id VARCHAR2, id2 VARCHAR2, Price2 VARCHAR2)
as

Begin 
  update meals Set PRICE = Price2
  Where  MT_ID= id and M_ID = id2 ;
END;
