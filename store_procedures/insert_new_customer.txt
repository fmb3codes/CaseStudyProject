CREATE OR REPLACE PROCEDURE SP_INS_CUST(ID INTEGER, FNAME varchar2, LNAME varchar2, 
                                      UPASSWORD varchar2, ULAST_LOGIN timestamp, UEMAIL varchar2, 
                                      UHOME_NUMBER NUMBER, UMOBILE_NUMBER NUMBER)
AS
BEGIN
  INSERT INTO CUSTOMERS(C_ID, FIRST_NAME, LAST_NAME, PASSWORD, LAST_LOGIN, EMAIL, HOME_NUMBER, MOBILE_NUMBER)
    VALUES(ID, FNAME, LNAME, UPASSWORD, ULAST_LOGIN, UEMAIL, UHOME_NUMBER, UMOBILE_NUMBER);

END;
