CREATE OR REPLACE PROCEDURE SP_INS_MEAL(MDESCRIPTION varchar2, MNAME varchar2, 
                                      MPRICE DECIMAL, MIMAGE_PATH varchar2, MMEAL_OF_DAY varchar2, 
                                      MMT_ID varchar2)
AS
BEGIN
  INSERT INTO MEALS(M_ID, DESCRIPTION, NAME, PRICE, IMAGE_PATH, MEAL_OF_DAY, MT_ID)
    VALUES(SEQ_CUSTOMERS.NEXTVAL, MDESCRIPTION, MNAME, MPRICE, MIMAGE_PATH, TO_DATE(MMEAL_OF_DAY, 'dd-MM-yyyy'), MMT_ID);

END;
